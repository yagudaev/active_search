def read_multi_entries(keys, **options)
  values = {}

  keys.each do |name|
    entry = read_entry(name, **options)
    values[name] = entry.value if entry
  end

  values
end

def write_entry(key, entry, **options)
  entry.dup_value!
  @data[key] = entry
  true
